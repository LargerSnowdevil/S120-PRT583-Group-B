public interface Cake {
   void bake();
}


public class Chocolate implements Cake {

   public void bake() {
  
   }
}
public class Vanilla implements Cake {

   public void bake() {
  
   }
}


public class CakeFactory {
   public static CakeFactory getFactory(string cakeType){   
      if(cakeType.equalsIgnoreCase("Chocolate")){
         return new ChocolateCakeFactory();         
      }else if(cakeType.equalsIgnoreCase("vanilla")) {
         return new VanillaCakeFactory();
      }
      return null;
   }
}

public class BakeCake {
   public static void main(String[] args) {

      CakeFactory CakeFactory = CakeFactory.getFactory(Vanilla);
    
      CakeFactory bake();
   
      CakeFactory CakeFactory1 = CakeFactory.getFactory(chocolate);

      CakeFactory1 bake();

      
   }
}
